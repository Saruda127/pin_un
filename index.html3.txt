    <script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>
    
    <script type="text/javascript">
        // More API functions here:
        // https://github.com/googlecreativelab/teachablemachine-community/tree/master/libraries/image

        // the link to your model provided by Teachable Machine export panel
        // ** (ใช้ URL เดิมของคุณ) **
        const URL = "https://teachablemachine.withgoogle.com/models/OX00UXnGI/";

        let model, webcam, labelContainer, maxPredictions;

        // Load the image model and setup the webcam
        async function init() {
            // ปรับแต่ง UI: ซ่อนปุ่ม, แสดงพื้นที่เว็บแคม
            document.getElementById('start-btn').classList.add('hidden');
            
            const modelURL = URL + "model.json";
            const metadataURL = URL + "metadata.json";

            // load the model and metadata
            model = await tmImage.load(modelURL, metadataURL);
            maxPredictions = model.getTotalClasses();

            // Convenience function to setup a webcam
            const flip = true; // whether to flip the webcam
            webcam = new tmImage.Webcam(200, 200, flip); // width, height, flip
            await webcam.setup(); // request access to the webcam
            await webcam.play();
            window.requestAnimationFrame(loop);

            // append elements to the DOM
            const webcamContainer = document.getElementById("webcam-container");
            webcamContainer.appendChild(webcam.canvas);
            webcamContainer.classList.add('loaded'); // ปรับแต่ง UI: ใส่ขอบและขยายพื้นที่

            labelContainer = document.getElementById("label-container");
            
            // ปรับปรุงการสร้าง Element เพื่อความสวยงาม (สร้างโครงสร้าง Progress Bar)
            for (let i = 0; i < maxPredictions; i++) {
                // สร้างแถว container
                const row = document.createElement("div");
                row.className = "prediction-row";

                // สร้าง Text แสดงชื่อ Class
                const labelSpan = document.createElement("span");
                labelSpan.className = "class-label";
                
                // สร้าง Container ของ Progress Bar
                const progressCont = document.createElement("div");
                progressCont.className = "progress-bar-container";
                
                // สร้างตัว Fill สี
                const progressFill = document.createElement("div");
                progressFill.className = "progress-bar-fill";
                progressCont.appendChild(progressFill);

                // สร้าง Text แสดง %
                const probSpan = document.createElement("span");
                probSpan.className = "probability-text";

                // ประกอบร่าง
                row.appendChild(labelSpan);
                row.appendChild(progressCont);
                row.appendChild(probSpan);
                
                labelContainer.appendChild(row);
            }
        }

        async function loop() {
            webcam.update(); // update the webcam frame
            await predict();
            window.requestAnimationFrame(loop);
        }

        // run the webcam image through the image model
        async function predict() {
            // predict can take in an image, video or canvas html element
            const prediction = await model.predict(webcam.canvas);
            
            for (let i = 0; i < maxPredictions; i++) {
                const classByName = prediction[i].className;
                const probability = prediction[i].probability;
                const percentage = (probability * 100).toFixed(0); // ทำเป็น % เลขกลมๆ

                // อ้างอิง Element ในแถวที่ i
                const currentRow = labelContainer.childNodes[i];
                const labelSpan = currentRow.querySelector('.class-label');
                const progressFill = currentRow.querySelector('.progress-bar-fill');
                const probSpan = currentRow.querySelector('.probability-text');

                // อัปเดตข้อมูล
                labelSpan.innerHTML = classByName;
                probSpan.innerHTML = percentage + "%";
                
                // อัปเดตความยาวแถบ Progress Bar
                progressFill.style.width = percentage + "%";

                // ปรับเปลี่ยนสีแถบตามความมั่นใจ (Dynamic Coloring)
                progressFill.className = "progress-bar-fill"; // reset class
                if(probability > 0.8) {
                    progressFill.classList.add('high-confidence');
                } else if (probability > 0.4) {
                    progressFill.classList.add('med-confidence');
                } else {
                    progressFill.classList.add('low-confidence');
                }
            }
        }
